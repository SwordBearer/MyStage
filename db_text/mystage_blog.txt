CREATE TABLE mystage_blog (
  id int(10) NOT NULL AUTO_INCREMENT,
  status int(4) NOT NULL DEFAULT '2',
  inputtime datetime NOT NULL,
  updatetime datetime NOT NULL,
  title varchar(100) NOT NULL,
  content text NOT NULL,
  catid int(10) DEFAULT '0',
  typeid int(10) NOT NULL DEFAULT '1',
  PRIMARY KEY (id),
  KEY blog_cat_cid (catid),
  CONSTRAINT blog_cat_cid FOREIGN KEY (catid) REFERENCES mystage_blog_cat (id) ON DELETE SET NULL ON UPDATE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=31 DEFAULT CHARSET=utf8;
